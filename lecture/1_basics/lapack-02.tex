  \item 例2.5.1 (ハンドブック2.5節)
\begin{lstlisting}
#include <stdio.h>
int main() {
  int q = 200;
  int* p = &q;
  printf("q is %d and *p is %d.\n", q, *p);
  return 0;
}
\end{lstlisting}
\begin{itemize}
\item \verb+q+のアドレスを\verb+p+に代入
\item \verb+q+と\verb+*p+の値を出力 → 両者とも200
\end{itemize}
